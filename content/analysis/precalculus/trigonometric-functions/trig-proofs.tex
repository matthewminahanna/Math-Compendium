\begin{exercise}
    Show that 
    \[ \frac{ \sec^{2}{ \left( \frac{x}{2} \right) }}{ \left( 1+ \tan{ \left( \frac{x}{2} \right) } \right)^{2}} = \frac{1}{1+ \sin{ \left( x \right) }} \]
\end{exercise}
\begin{solution}
    \begin{align*}
        \frac{ \sec^{2}{ \left( \frac{x}{2} \right) }}{ \left( 1+ \tan{ \left( \frac{x}{2} \right) } \right)^{2}} &= \frac{ \sec^{2}{ \left( \frac{x}{2} \right) }}{1 + 2 \tan{ \left( \frac{x}{2} \right) } + \tan^{2}{ \left( \frac{x}{2} \right) }} \\
    &= \frac{\sec^{2}{ \left( \frac{x}{2} \right) }}{\sec^{2}{ \left( \frac{x}{2} \right) } + 2 \tan{ \left( \frac{x}{2} \right) }} \\
     &= \frac{\sec^{2}{ \left( \frac{x}{2} \right) }}{\sec^{2}{ \left( \frac{x}{2} \right) } + 2 \tan{ \left( \frac{x}{2} \right) }} \cdot \frac{ \cos{^{2} \left( \frac{x}{2} \right) }}{\cos{^{2} \left( \frac{x}{2} \right) }} \tag{Justified since we never multiply by $\frac{0}{0}$ } \\
     &= \frac{1}{1 + 2 \sin{ \left( \frac{x}{2} \right) } \cos{ \left( \frac{x}{2} \right) }} \\
     &= \frac{1}{1+ \sin{ \left(x \right) }}
    \end{align*}
    
\end{solution}
