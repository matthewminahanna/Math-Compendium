\begin{dfn}\label{def:sigma-algebra}
    Suppose that \( X \) is a set \( \mathscr{S} \subseteq 2^{X} \). We call \( \mathscr{S} \) a \vocab{\( \sigma \)-algebra on \( X \)}  if
    \begin{enumerate}[label=\textbf{\roman*)}]
        \item \( \varnothing \in \mathscr{S}\)
        \item if \( E \in \mathscr{S} \), then \( X-E \in \mathscr{S} \) 
        \item if \( \left\{ E_{k} \right\}_{k=1}^{\infty} \subseteq \mathscr{S} \), then \( \bigcup_{k=1}^{\infty} E_{k} \in \mathscr{S} \)
    \end{enumerate}
\end{dfn}
