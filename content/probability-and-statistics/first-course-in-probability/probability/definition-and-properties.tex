\begin{dfn}
    The set \( \Omega \) of all possible outcomes of an experiment is called the \vocab{sample space}.  
    A subset \( A \subseteq \Omega \) is called an \vocab{event}.
\end{dfn}

\begin{example}
    Suppose that we conduct an experiment of rolling a six-sided die. Then 
    \[ \Omega = \{\text{roll a }1,\text{ roll a }2,\dots,\text{ roll a }6\},\]
    An example event is "roll a \( 1 \) or a \( 4 \).
\end{example}


\begin{dfn}
    A \vocab{probability space} is a triplet \( (\Omega, \mathscr{F}, \mathbb{P}) \) where
    \begin{enumerate}[label=\textbf{\roman*)}]
        \item \( \Omega \) is the sample space,
        \item \( \mathscr{F} \) is a \hyperref[def:sigma-algebra]{\( \sigma \)-algebra on \( \Omega \)} whose elements are events,
        \item \( \mathbb{P} : \mathscr{F} \to [0,1] \) is a \vocab{probability measure} or just \vocab{probability}. 
    \end{enumerate}
    The function \( \mathbb{P} \) satisfies:
    \begin{enumerate}[label=\textbf{\roman*)}]
        \setcounter{enumi}{3}
        \item \( \mathbb{P}(\Omega) = 1 \),
        \item If \( A_1, A_2, \dots \) are pairwise disjoint events, then
        \[
        \mathbb{P}\!\left( \bigcup_{k=1}^{\infty} A_k \right)
        = \sum_{k=1}^{\infty} \mathbb{P}(A_k).
        \]
    \end{enumerate}
    We read \( \mathbb{P}(A) \) as “the probability that \( A \) occurs.”
\end{dfn}



If you are not yet familiar with the formal definition of a \( \sigma \)-algebra, do not worry.  
For our purposes, it is enough to know that it is a collection of events with the following properties:

\begin{enumerate}[label=\textbf{\roman*)}]
    \item The impossible event \( \varnothing \) is included.
    \item The certain event \( \Omega \) is included.
    \item If \( A \) is an event, then its complement \( A^c \) (the event that \( A \) does not occur) is also an event.
    \item If \( A_1, A_2, \dots \) are events, then their union \( \bigcup_{k=1}^{\infty} A_k \) is also an event.
\end{enumerate}

These properties ensure that we can consistently assign probabilities to events and combinations of events.

\begin{lemma}
    \( \mathbb{P} \left( \varnothing \right)=0 \).
\end{lemma}
\begin{proof}
    Since \( \Omega \cap \varnothing = \varnothing \), they are pairwise disjoint. Then
    \begin{align*}
        1 &= \mathbb{P} \left( \Omega \right)\\ 
        &= \mathbb{P} \left( \Omega \cup \varnothing \right) \\
        &= \mathbb{P} \left( \Omega \right) + \mathbb{P} \left( \varnothing \right) \\
       1 &= 1 + \mathbb{P} \left( \varnothing \right)
    \end{align*}
    So  \( \mathbb{P} \left( \varnothing \right)=0 \).
\end{proof}

\begin{lemma}
    If \( A^{c} \) denotes \( \Omega - A \) where \( A \in \mathscr{F} \). Then 
    \[ \mathbb{P} \left( A^{c} \right) = 1- \mathbb{P} \left( A \right) .\]
\end{lemma}
\begin{proof}
    Very similar to above (In fact, the previous result is just a specific case of this result.) \( A^{c} \cap A = \varnothing \) and \( A^{c} \cup A = \Omega \) 
    \begin{align*}
        1 &= \mathbb{P} \left( \Omega \right)\\ 
        &= \mathbb{P} \left( A \cup A^{c} \right) \\
        &= \mathbb{P} \left( A \right) + \mathbb{P} \left( A^{c} \right)
    \end{align*}
    So \( \mathbb{P} \left( A^{c} \right) = 1- \mathbb{P} \left( A \right) \).
\end{proof}

\begin{lemma}\label{thm:inequality-for-subsets-of-prob-space}
    If \( A \) and \( B \) are events such that \( A \subseteq B \), then \( \mathbb{P} \left( A \right) \le \mathbb{P} \left( B \right)\).
\end{lemma}
\begin{proof}
    If \( A \subseteq B \), then we can write 
    \[ B = A \cup \left( B \cap A^{c} \right) .\] So 
    \begin{align*}
        \mathbb{P} \left( B \right) &=  \mathbb{P} \left( A \cup \left( B \cap A^{c} \right)  \right) \\
        \mathbb{P} \left( B \right)&= \mathbb{P} \left( A \right) + \mathbb{P} \left( B \cap A^{c} \right) \tag*{Since $A$ and $B \cap A^{c}$ are disjoint.}
    \end{align*}
    Since \( \mathbb{P} \left( B \cap A^{c} \right) \) is non-negative, it follows that \( \mathbb{P} \left( A \right) \le \mathbb{P} \left( B \right)\).
\end{proof}

\begin{corollary}
    For any event \( A \), \( \mathbb{P} \left( A \right) \le 1 \).
\end{corollary}
\begin{proof}
    Since \( \mathbb{P} \left( \Omega \right) =1 \), the result follows by \cref{thm:inequality-for-subsets-of-prob-space}.
\end{proof}

\begin{theorem}\label{thm:P-of-A-union-B}
    For any events \( A \) and \( B \), we have 
    \[ \mathbb{P} \left( A \cup B \right) = \mathbb{P} \left( A \right) + \mathbb{P} \left( B \right) - \mathbb{P} \left( A \cap B \right). \]
\end{theorem}
\begin{proof}
    We can perform a separation of \( A \cup B \) as all of the elements of \( A \) union all the elements of \( B \) that are not in \( A \) or 
    \[ A \cup B = A \cup \left( A^{c} \cap B \right) .\]
    This gives us 
    \begin{equation}\label{eqn:P-of-A-or-B}
        \mathbb{P} \left( A \cup B \right) = \mathbb{P} \left( A \right) + \mathbb{P} \left( A^{c} \cap B \right)
    \end{equation}
    Similarly, we can write \( B \) as the collection of all elements of \( B \) that are in \( A \) union all the elements of \( B \) that are not in \( A^{c} \) or 
    \[ B = \left( A \cap B \right) \cup \left( A^{c} \cap B \right) .\]
    So 
    \[ \mathbb{P} \left( B \right) = \mathbb{P} \left( A \cap B \right) + \mathbb{P} \left( A^{c} \cap B \right) \]
    or 
    \begin{equation}\label{eqn:P-of-A-complement-and-B}
        \mathbb{P} \left( A^{c} \cap B \right) = \mathbb{P} \left( B \right) - \mathbb{P} \left( A \cap B \right) 
    \end{equation}
    Substituting \cref{eqn:P-of-A-complement-and-B} into \cref{eqn:P-of-A-or-B} establishes the result.
\end{proof}

\begin{theorem}
Let \( X = \left\{ X_{1}, X_{2}, \dots, X_{n} \right\} \) be a collection of events. Then 
\[ \mathbb{P} \left( \bigcup_{k=1}^{n} X_{k} \right) = \sum_{Y \in 2^{X}} (-1)^{\abs{Y}+1} \;  \; \mathbb{P} \left( \bigcap_{X_{k} \in Y} X_{k} \right)\] where \( 2^{X} \) is the power set of \( X \).
\end{theorem}
\begin{proof}
    We will prove this by induction. Note that \cref{thm:P-of-A-union-B} is the base case so we can proceed immediately to the inductive step. \\ 
    Suppose, for induction, that the result holds for all collections of n events. We pick a particular collection \( X = \left\{  X_{1}, X_{2}, \dots, X_{n}\right\} \) and consider \( X' = X \cup \left\{ X_{n+1} \right\} \). Then 
    \begin{align*}
        \mathbb{P} \left( \bigcup_{k=1}^{n+1} X_{k}\right) &= \mathbb{P} \left(   \left(  \bigcup_{k=1}^{n} X_{k}\right) \cup X_{n+1} \right) \\
        &=\mathbb{P} \left( \bigcup_{k=1}^{n} X_{k} \right) + \mathbb{P} \left( X_{n+1} \right) - \mathbb{P} \left(   \left(  \bigcup_{k=1}^{n} X_{k}\right) \cap X_{n+1} \right) \tag*{By the base case.} \\
        &= \mathbb{P} \left( \bigcup_{k=1}^{n} X_{k} \right) + \mathbb{P} \left( X_{n+1} \right) - \mathbb{P} \left( \bigcup_{k=1}^{n} \left( X_{k} \cap X_{n+1} \right) \right)
    \end{align*}
    Now if we let \( \overline{X} = \left\{ X_{k} \cap X_{n+1} \right\}_{k=1}^n \), we have 
    \begin{align*}
         \mathbb{P} \left( \bigcup_{k=1}^{n+1} X_{k}\right) &= \mathbb{P} \left( \bigcup_{k=1}^{n} X_{k} \right) + \mathbb{P} \left( X_{n+1} \right) - \mathbb{P} \left( \bigcup_{k=1}^{n} \left( X_{k} \cap X_{n+1} \right) \right) \\
         &= \mathbb{P} \left( \bigcup_{k=1}^{n} X_{k} \right) + \mathbb{P} \left( X_{n+1} \right) - \left[ \sum_{Y \in 2^{\overline{X}}} \left( -1 \right)^{\abs{Y}+1} \; \; \mathbb{P} \left( \bigcap_{X_{k} \in Y} X_{k} \right) \right] \\
         &= \sum_{Y \in 2^{X}} (-1)^{\abs{Y}+1} \;  \; \mathbb{P} \left( \bigcap_{X_{k} \in Y} X_{k} \right) + \mathbb{P} \left( X_{n+1} \right) +  \sum_{Y \in 2^{\overline{X}}} \left( -1 \right)^{\abs{Y}} \; \; \mathbb{P} \left( \bigcap_{X_{k} \in Y} X_{k} \right)
    \end{align*}
   The above sum can be re-indexed to be 
   \[  \mathbb{P} \left( \bigcup_{k=1}^{n+1} X_{k}\right) = \sum_{Y \in 2^{X'}} (-1)^{\abs{Y}+1} \;  \; \mathbb{P} \left( \bigcap_{X_{k} \in Y} X_{k} \right) \]
   as desired. 
\end{proof}
