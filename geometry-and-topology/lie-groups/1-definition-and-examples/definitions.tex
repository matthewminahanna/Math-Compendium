\begin{dfn}
    A \vocab{topological group} is a group \( G \) equipped with a topology such that the map
    \[
        G \times G \to G, \quad (g,h) \mapsto g h^{-1}
    \]
    is continuous (where \( G \times G \) carries the product topology).
\end{dfn}


\begin{lemma}
    A group \( G \) is a topological group if and only if the maps 
    \[ G \times G \to G, \quad \left( g,h \right) \mapsto gh \]
    and 
    \[ G \to G, \quad g \mapsto g^{-1} \]
    are continuous. 
\end{lemma}
\begin{proof}
    \( \Rightarrow \) Suppose \( G \) is a topological group. Then the map
    \[
        f \colon G \times G \to G, \quad (g, h) \mapsto gh^{-1}
    \]
    is continuous by definition.

    Define the embedding map \( \mathrm{emb} \colon G \to G \times G \) by
    \[
        \mathrm{emb}(g) := (e, g),
    \]
    where \( e \) is the identity element of \( G \). This map is continuous, since it is the product of the constant map \( g \mapsto e \) and the identity map \( g \mapsto g \).

    Then the composition
    \[
        G \xrightarrow{\mathrm{emb}} G \times G \xrightarrow{f} G,
    \]
    given by \( g \mapsto f(e, g) = eg^{-1} = g^{-1} \), is continuous. Hence, inversion \( g \mapsto g^{-1} \) is continuous.

  Now we show that the multiplication map \( G \times G \to G \), given by \( (g, h) \mapsto gh \), is continuous.

Consider the composition of the following two continuous maps:
\begin{align*}
    G \times G &\xrightarrow{\mathrm{id} \times \mathrm{inv}} G \times G, \quad (g, h) \mapsto (g, h^{-1}), \\
    G \times G &\xrightarrow{f} G, \quad (g, h) \mapsto gh^{-1},
\end{align*}
where \( \mathrm{inv}(h) = h^{-1} \), and \( f(g, h) = gh^{-1} \).

Then their composition is:
\[
    (g, h) \mapsto f(g, h^{-1}) = g(h^{-1})^{-1} = gh.
\]
Since both maps are continuous, the composition is continuous. Hence the multiplication map is continuous.


\end{proof}
