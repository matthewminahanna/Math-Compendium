\begin{dfn}
    The vectors \( \vb{v}_{1}, \vb{v}_{2}, \dots, \vb{v}_{n} \) are said to be \vocab{linearly independent} if there is only one solution to the equation 
    \[ \sum_{k=1}^{n} \lambda_{k} \vb{v}_{k} = \vb{0}  \]
    namely, \( \lambda_{1}= \lambda_{2} = \cdots = \lambda_{n} = 0 \).
\end{dfn}
